function setCookieStorage(cookieName, cookieValue)
{

function getCookieStorage(cookieName)
{
    var regexp = new RegExp("(?:(?:^|.*;\s*)" + cookieName + "\s*\=\s*([^;]*).*$)|^.*$", "g");
    
    return document.cookie.replace(regexp, "$1");
}
